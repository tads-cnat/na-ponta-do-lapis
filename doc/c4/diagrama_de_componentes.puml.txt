' Força o layout a ir da esquerda para a direita (horizontal)
LAYOUT_LEFT_RIGHT()

title Diagrama de Componentes Horizontal — Na Ponta do Lápis

' ─── Atores externos ───
Person(user, "Usuário", "Pessoa que gerencia suas finanças pessoais")

' ─── Sistemas externos ───
System_Ext(awesomeApi, "AwesomeAPI", "Fornece cotações de moedas em tempo real via HTTP")

' ─── Container: Web App ───
Container(webApp, "Web App", "Angular", "Interface web para acesso ao sistema de finanças")


' ─── Container: API ───
Container(api, "API NPL", "Spring Boot", "Processa as regras de negócio e gerencia o sistema")

' ─── Banco de dados ───
ContainerDb(db, "Banco de Dados", "PostgreSQL", "Armazena usuários, contas, transações, metas e configurações")

' ─── Fluxo de Integração ───
Rel(user,   webApp,  "Acessa via navegador", "HTTPS")
Rel(webApp, api, "Faz chamadas REST", "JSON/HTTPS")

' Mudado para Rel_R (Relation Right) para forçar o fluxo horizontal em direção ao banco e API externa
Rel_R(api, db, "Lê e escreve dados", "ORM / JDBC")
Rel_R(api, awesomeApi, "Consulta cotação de moedas", "JSON/HTTPS")

@enduml
